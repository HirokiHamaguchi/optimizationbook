\section*{Typos}
\begin{itemize}
    \item \textbf{P.7} In Sec.~1.1, the region where $f$ is not $\epsilon$ has volume \red{$\frac{1}{\epsilon^n}$} times the volume of the unit ball.
          \begin{itemize}
              \item This should be \blue{$\epsilon^n$}.
          \end{itemize}

    \item \textbf{P.12} In Sec.~1.3.3, there are some \red{$x^i$}.
          \begin{itemize}
              \item This should be \blue{$x_i$}.
          \end{itemize}

    \item \textbf{P.13} In Def.~1.14, the epigraph of $f : \mathbb{R}^n \to$ \red{$\mathbb{R}$} is $\mathrm{epi} f \coloneqq \{(x,t) \in \mathbb{R}^n \times \mathbb{R} : t \geq f(x)\}$.
          \begin{itemize}
              \item This should be \blue{$\mathbb{R} \cup \{+\infty\}$}.
          \end{itemize}

    \item \textbf{P.13} In Ex.~1.18, the function $h(x) =$ \red{$\max f(x), g(x)$} is also convex.
          \begin{itemize}
              \item This should be \blue{$\max\{f(x), g(x)\}$}.
          \end{itemize}

    \item \textbf{P.15} In Ex.~1.21, there are some \red{$x^i$} and \red{$y^i$}.
          \begin{itemize}
              \item This should be \blue{$x_i$} and \blue{$y_i$}.
          \end{itemize}
\end{itemize}
